Getting started

Installation

via npm

npm install vue-swatches --save

via yarn

yarn add vue-swatches

via CDN

<script src="https://unpkg.com/vue-swatches"></script>

Basic usage

<!-- Vue component -->
<template>
  <div>
    <swatches v-model="color" />
  </div>
</template>

<script>
  import Swatches from 'vue-swatches'

  // Register globally
  Vue.component('swatches', Swatches)

  export default {
    // or register locally
    components: { Swatches },
    data () {
      return {
        color: '#1CA085'
      }
    }
  }
</script>
